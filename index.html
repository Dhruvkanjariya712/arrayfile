                    //call by value

                    #include

                        a = a + 100;

                        void main()
                        {
                        int a = 100;

                        }
            
                    //call by value(swaping of two numbers)
            
                    #include 
                        void swap(int a, int b)
                        {
                        a = a + b;
                        b = a - b;
                        a = a - b;
                        printf("a=%d , b=%d", a, b);
                        }
                        void main()
                        {
                        int a, b;
                        printf("enter the value of a & b\n");

                    //call by reference
        
                    #include 
                        void change(int *a)
                        {
                        printf("before adding value inside function num=%d\n", *a);
                        *a += 100;
                        printf("after adding value inside function num=%d\n", *a);
                        }
                        void main()
                        {
                        int a = 100;
                        printf("before function call a=%d\n", a);
                        change(&a);
                        printf("after function call a=%d\n", a);
                        }
        
                    //call by reference(swaping of two numbers)
        
                    #include 
                        void swap(int *a, int *b)
                        {
                        *a = *a + *b;
                        *b = *a - *b;
                        *a = *a - *b;
                        printf("after swapping\n");
                        printf("a=%d , b=%d", *a, *b);
                        }
                        void main()
                        {
                        int a, b;
                        printf("enter the value of a & b\n");
                        scanf("%d %d", &a, &b);
                        swap(&a, &b);
                        }

Recursive function

            

                //addition of series in recursion

                #include 
                int add(int a)
                {
                    if (a == 0)
                    {
                        return 0;
                    }
                    else
                    {
                        return a + add(a - 1);
                    }
                }
                void main()
                {
                    int a;
                    printf("enter the number\n");
                    scanf("%d", &a);
                    printf("Addition of 0 to %d is %d", a, add(a));
                }
                
                //factorial in recursion
                
                #include 
                int add(int a)
                {
                    if (a == 0)
                    {
                        return 1;
                    }
                    else
                    {
                        return a * add(a - 1);
                    }
                }
                void main()
                {
                    int a;
                    printf("enter the number\n");
                    scanf("%d", &a);
                    printf("factorial of  %d is %d", a, add(a));
                }
                
                //fibonaci with recurtion
                
                #include 
                int add(int a)
                {
                    if (a == 1)
                    {
                        return 1;
                    }
                    else if (a == 2)
                    {
                        return 1;
                    }
                
                    else
                    {
                        return add(a - 1) + add(a - 2);
                    }
                }
                void main()
                {
                    int a;
                    printf("enter the number\n");
                    scanf("%d", &a);
                    printf("%d", add(a));
                }

                    //function with pointer and array

                    #include 
                    int foo(int *p, int size);
                    void main()
                    {
                        int p[] = {1, 2, 3, 4, 5};
                        printf("%d", foo(p, 5));
                    }
                    int foo(int *p, int size)
                    {
                        int c = 0;
                        for (int i = 0; i < size; i++)
                        {
                            c += *p;
                            p++;
                        }
                        return c;
                    }
