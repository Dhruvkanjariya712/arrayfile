                //addition of series in recursion

                #include 
                int add(int a)
                {
                    if (a == 0)
                    {
                        return 0;
                    }
                    else
                    {
                        return a + add(a - 1);
                    }
                }
                void main()
                {
                    int a;
                    printf("enter the number\n");
                    scanf("%d", &a);
                    printf("Addition of 0 to %d is %d", a, add(a));
                }

                #include 
                int add(int a)
                {
                    if (a == 0)

                        return 1;

                    {

                    }

                //fibonaci with recurtion
                
                #include 
                int add(int a)
                {
                    if (a == 1)
                    {
                        return 1;
                    }
                    else if (a == 2)
                    {
                        return 1;
                    }
                
                    else
                    {
                        return add(a - 1) + add(a - 2);
                    }
                }
                void main()
                {
                    int a;
                    printf("enter the number\n");
                    scanf("%d", &a);
                    printf("%d", add(a));
                }
