Call by value & Call by reference

             call by value

                    #include 
                        void change(int a)
                        {
                        printf("before adding value inside function num=%d\n", a);
                        a = a + 100;
                        printf("after adding value inside function num=%d\n", a);
                        }
                        void main()
                        {
                        int a = 100;
                        printf("before function call a=%d\n", a);
                        change(a);
                        printf("after function call a=%d\n", a);
                        }
            
                    call by value(swaping of two numbers)
            
                    #include 
                        void swap(int a, int b)
                        {
                        a = a + b;
                        b = a - b;
                        a = a - b;
                        printf("a=%d , b=%d", a, b);
                        }
                        void main()
                        {
                        int a, b;
                        printf("enter the value of a & b\n");
                        scanf("%d %d", &a, &b);
                        swap(a, b);
                        }
        
                    call by reference
        
                    #include 
                        void change(int *a)
                        {
                        printf("before adding value inside function num=%d\n", *a);
                        *a += 100;
                        printf("after adding value inside function num=%d\n", *a);
                        }
                        void main()
                        {
                        int a = 100;
                        printf("before function call a=%d\n", a);
                        change(&a);
                        printf("after function call a=%d\n", a);
                        }
        
                    call by reference(swaping of two numbers)
        
                    #include 
                        void swap(int *a, int *b)
                        {
                        *a = *a + *b;
                        *b = *a - *b;
                        *a = *a - *b;
                        printf("after swapping\n");
                        printf("a=%d , b=%d", *a, *b);
                        }
                        void main()
                        {
                        int a, b;
                        printf("enter the value of a & b\n");
                        scanf("%d %d", &a, &b);
                        swap(&a, &b);
                        }
